// Left Join 
    -- “Give me ALL rows from the left table, even if no match exists on the right.” 

    syntax :
            SELECT * 
            FROM users 
            LEFT JOIN orders on users.id = orders.user_id 

            note : users without orders became "null"

// Right Join

    -- right join is the same idea as the left join which returns all the matched rows with the left side and also return if no rows matches for no row matches the left side field value became null

    syntax :
            SELECT customers.name , orders.total
            From customers
            RIGHT JOIN orders on customers.id = orders.customer_id;



// Inner Join
    -- give me only rows that exist in both tables

    syntax :
            SELECT * 
            FROM customers
            INNER JOIN orders customer.id = orders.customer_id;

            note: returns only when both rows matches 

// Full Outer Join

    -- returns every matched rows and even non matched rows from both side left side and right side and for non matching rows value became null .

    and it is not directly supported in sql but do supported in postgressSQL.

    * people often confuses full outer join with 

        1.Left + right join without nulls = full outer join which is incorrect
        2. full outer join is same as uniion incorrect.
        3. full outer join is for forieng keys also incorrect 

    * full outer join very rare in crud apps
    it is used for data reconcilation and reporting analytics and auditing
